static string met(string n) 
{ 
if (n.IndexOf("+") != -1) 
n = n.Replace("+","сложение"); 
if (n.IndexOf("-") != -1) 
n = n.Replace("-", "вычитание"); 
if (n.IndexOf("*") != -1) 
n = n.Replace("*", "ужножение"); 
if (n.IndexOf("/") != -1) 
n = n.Replace("/", "деление"); 
return n; 
} 

static void Main(string[] args) 
{ 
Console.WriteLine(met("1+2-3*4/5")); 
}
