using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace Нетребин_Владимир //10МИ2 Нетребин Владимир 01.02.2017 Вариант 7 (подсчет числа гласных букв каждой строки в массиве строк)
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("Введите количество строк: "); //пользователь вводит количество строк в массиве
            int n = int.Parse(Console.ReadLine());//считываем их количество
            string[] arr = new string [n];
            char [] s = {'a','A','e','E','y','Y','u','U','i','I','o','O'};
            int i = 0;
            int c = 0;
            int sum = 0;
            for (i = 0; i < n; i++)
            {   
                Console.WriteLine("Введите "+(i+1)+" строку: ");
                c = 0;               
                arr[i] = Console.ReadLine();//считываем строку
                char [] split = arr[i].ToCharArray();//преобразуем её в массив символов
                foreach (char sp in split)//считаем гласные в массиве символов
                {
                    if (s.Contains(sp))
                        c++;
                }
                sum += c;
                Console.WriteLine("Количество гласных в строке: " + c); //вывод количества гласных в строке
            }
            Console.WriteLine("Количество гласных в массиве: " + sum);//вывод количества гласных в массиве
            Console.ReadLine();
        }
       
    }
}
