using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace typeOfString
{
    class Program
    {
        public static string peredel(string str)
        {
            if (str.Contains("+"))
            {
                return "Сумма";
            }
            if (str.Contains("-"))
            {
                return "Вычитание";
            }
            if (str.Contains("*"))
            {
                return "Умножение";
            }
            if (str.Contains("/"))
            {
                return "Деление";
            }
            return "cheto не то";

        }
        static void Main(string[] args)
        {
            Console.WriteLine("Введите строку: ");
            string str = Console.ReadLine();
            Console.WriteLine(peredel(str));
            Console.ReadKey();
        }
    }
}
